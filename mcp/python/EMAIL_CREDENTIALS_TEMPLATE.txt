# Add these lines to your .env file to enable email notifications

# Email Configuration
ENABLE_EMAIL=true
SMTP_SERVER=smtp.gmail.com
SMTP_PORT=587
SENDER_EMAIL=your-gmail@gmail.com
SENDER_PASSWORD=xxxx xxxx xxxx xxxx

# Instructions:
# 1. Replace 'your-gmail@gmail.com' with your actual Gmail address
# 2. For SENDER_PASSWORD, use App Password from: https://myaccount.google.com/apppasswords
#    - Go to the link
#    - Select "Mail" and "Windows Computer"
#    - Google will generate 16 characters - paste them (with spaces) as SENDER_PASSWORD
# 3. Save to .env
# 4. Run: python3 test_email_notifier.py
